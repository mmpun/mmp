#include<stdio.h>
int main()
{
	int a[50];
	int rt[10]; //for storing retrieval time
	int l,i,j,n,temp,sum=0,psum=0;
	printf("enter the no. of programs"); //n is no of program that can be stored on the tape 
	scanf("%d",&n);
	printf("enter the length of programs\n");
	
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	

	printf("original program length ");
	for(i=0;i<n;i++)
	{
		printf("%d\t",a[i]);
	}
	
	for(i=0;i<n-1;i++)
	{
		for(j=0;j<n-1;j++)
		{
			if(a[j+1]<a[j])
			{
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
	printf("sorted program length ");

	for(i=0;i<n;i++)
	{
		printf("%d\t",a[i]);
	}
	for(i=0;i<n;i++)
	{

		psum=psum+a[i];
		sum=psum+sum;
		rt[i]=sum;
	}
	printf("\ntotal retrieval time is %d\n",rt[n-1]);
	printf("mean retrieval time is %f\n",rt[n-1]/(float)n);
	return 0;
}
